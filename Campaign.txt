public class Campaign {

	private String campaignName;
	
	public Campaign() {

	}
	
	public Campaign(String campaignName) {
		this.campaignName = campaignName;
	}

		return campaignName;
	}

	public void setCampaignName(String campaignName) {
		this.campaignName = campaignName;
	}
	
	
}
